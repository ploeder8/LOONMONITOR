Datamodel

PostgreSQL 16+ datamodel met bitemporeel patroon, encryptie van persoonsgegevens, idempotente runs en append-only audit-trail.

Scope: dit datamodel is voor de volledige loonmotor (productie-database). De Jaakie POC gebruikt een statische JSON-dataset in de browser (src/data/pc200_payroll_dataset_2026.json) — zie Jaakie-specificatie §5.

1. Ontwerpprincipes

  • Bitemporeel — twee tijdassen: valid_period (geldigheid bedrijfsmatig) + system_period (database-versie). Maakt retroactieve correcties mogelijk zonder geschiedenis te verliezen. MVP
  • Multi-tenant — elke tabel met persoonsgegevens heeft tenant_id. Row-Level Security (RLS) filtert per tenant in elke query.
  • Monetair — numeric(18,4), BigDecimal in Kotlin, afronding HALF_UP op laatste 2 decimalen bij output.
  • Encryptie — RRN, IBAN en gevoelige PII versleuteld at-rest met kolom-niveau encryptie (pgcrypto + KMS-managed sleutels).
  • Idempotency — elke PayrollRun heeft canonical_input_hash en rule_set_version. Zelfde input + zelfde regels → zelfde output.
  • Audit-trail — append-only tabel met SHA-256 ketenhash. Niet uit te wissen.

3. SQL DDL — kernentiteiten

tenant

CREATE TABLE tenant (
  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  naam            text NOT NULL,
  kbo_nummer      varchar(10),
  status          text NOT NULL CHECK (status IN ('actief','opgeschort','beëindigd')),
  created_at      timestamptz NOT NULL DEFAULT now(),
  system_period   tstzrange NOT NULL DEFAULT tstzrange(now(), null, '[)')
);

werknemer (met encryptie RRN)

CREATE TABLE werknemer (
  id               uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id        uuid NOT NULL REFERENCES tenant(id),
  werkgever_id     uuid NOT NULL,
  rrn_encrypted    bytea NOT NULL,      -- pgcrypto / KMS
  rrn_hash         bytea NOT NULL,      -- HMAC voor unique-check zonder decrypt
  voornaam         text NOT NULL,
  achternaam       text NOT NULL,
  geboortedatum    date NOT NULL,
  geslacht         text CHECK (geslacht IN ('M','V','X')),
  nationaliteit    text,
  adres            jsonb,
  iban_encrypted   bytea,
  created_at       timestamptz NOT NULL DEFAULT now(),
  UNIQUE (tenant_id, rrn_hash)
);
CREATE INDEX werknemer_werkgever_idx ON werknemer (tenant_id, werkgever_id);

werknemer_gezinssituatie (bitemporeel)

CREATE TABLE werknemer_gezinssituatie (
  id                uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  werknemer_id      uuid NOT NULL REFERENCES werknemer(id),
  burgerlijke_staat text NOT NULL,
  partner_inkomen   text CHECK (partner_inkomen IN ('geen','laag','hoog','nvt')),
  kinderen_ten_laste smallint NOT NULL DEFAULT 0,
  andere_ten_laste  smallint NOT NULL DEFAULT 0,
  handicap_eigen    boolean NOT NULL DEFAULT false,
  handicap_partner  boolean NOT NULL DEFAULT false,
  valid_period      daterange NOT NULL,
  system_period     tstzrange NOT NULL DEFAULT tstzrange(now(), null, '[)'),
  EXCLUDE USING gist (werknemer_id WITH =, valid_period WITH &&,
                      system_period WITH &&)
);

De EXCLUDE constraint voorkomt overlappende geldigheidsperiodes binnen dezelfde systeemversie.

contract

CREATE TABLE contract (
  id                uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id         uuid NOT NULL REFERENCES tenant(id),
  werknemer_id      uuid NOT NULL REFERENCES werknemer(id),
  werkgever_id      uuid NOT NULL,
  type              text NOT NULL CHECK (type IN
                       ('OBD','BD','vervanging','student','flexi','leerling','andere')),
  statuut           text NOT NULL CHECK (statuut IN ('bediende','arbeider','andere')),
  start_datum       date NOT NULL,
  einde_datum       date,
  opzeg_einde       date,
  paritair_comite   varchar(10) NOT NULL,
  arbeidsregime_uren_per_week numeric(5,2) NOT NULL,
  bruto_maandloon   numeric(12,2),
  uurloon           numeric(12,4),
  barema_code       text,
  functie           text,
  anciënniteitsdatum date,
  created_at        timestamptz NOT NULL DEFAULT now()
);

loonperiode & payrollrun (idempotent)

CREATE TABLE loonperiode (
  id            uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id     uuid NOT NULL REFERENCES tenant(id),
  werkgever_id  uuid NOT NULL,
  periode       daterange NOT NULL,
  status        text NOT NULL CHECK (status IN ('open','berekend','goedgekeurd','afgesloten')),
  UNIQUE (werkgever_id, periode)
);

CREATE TABLE payrollrun (
  id                    uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id             uuid NOT NULL REFERENCES tenant(id),
  loonperiode_id        uuid NOT NULL REFERENCES loonperiode(id),
  rule_set_version      text NOT NULL,
  canonical_input_hash  bytea NOT NULL,    -- SHA-256 over canonical_json(input)
  triggered_by          uuid NOT NULL,
  triggered_at          timestamptz NOT NULL DEFAULT now(),
  status                text NOT NULL CHECK (status IN ('running','success','failed','superseded')),
  UNIQUE (loonperiode_id, canonical_input_hash, rule_set_version)
);

De UNIQUE constraint maakt herberekening met dezelfde input een no-op — krachtige idempotency op het runlevel.

berekeningsresultaat (met explain)

CREATE TABLE berekeningsresultaat (
  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id       uuid NOT NULL REFERENCES tenant(id),
  payrollrun_id   uuid NOT NULL REFERENCES payrollrun(id),
  werknemer_id    uuid NOT NULL REFERENCES werknemer(id),
  bruto           numeric(18,4) NOT NULL,
  rsz_werknemer   numeric(18,4) NOT NULL,
  belastbaar      numeric(18,4) NOT NULL,
  bv              numeric(18,4) NOT NULL,
  bbsz            numeric(18,4) NOT NULL,
  inhoudingen     numeric(18,4) NOT NULL,
  beslag          numeric(18,4) NOT NULL,
  netto           numeric(18,4) NOT NULL,
  rsz_werkgever   numeric(18,4) NOT NULL,
  totale_wg_kost  numeric(18,4) NOT NULL,
  explain         jsonb NOT NULL,             -- "audit trail" per regel
  UNIQUE (payrollrun_id, werknemer_id)
);

looncomponent_definitie (regelmotor-data)

CREATE TABLE looncomponent_definitie (
  code              text PRIMARY KEY,
  omschrijving      text NOT NULL,
  rsz_plichtig      boolean NOT NULL,
  bv_plichtig       boolean NOT NULL,
  belastbaar        boolean NOT NULL,
  basis_voor_vakantiegeld boolean NOT NULL DEFAULT false,
  basis_voor_eindejaarspremie boolean NOT NULL DEFAULT false,
  exceptionele_bezoldiging boolean NOT NULL DEFAULT false,
  valid_period      daterange NOT NULL,
  bron_juridisch    text,                     -- bv. 'KB 27/9/1966'
  rule_set_version  text NOT NULL
);

regelversie (parameter-pakket)

CREATE TABLE regelversie (
  versie         text PRIMARY KEY,
  publicatie     date NOT NULL,
  bron           text NOT NULL,                  -- bv. KB 11/12/2025
  changelog      text,
  payload        jsonb NOT NULL                  -- decision tables, schaaltabellen
);

Elke payrollrun verwijst naar exact één regelversie. Een correctie van een lopende periode draait altijd met de regelversie die toen geldig was.

audit_log (tamper-evident)

CREATE TABLE audit_log (
  id            bigserial PRIMARY KEY,
  tenant_id     uuid,
  actor_id      uuid,
  action        text NOT NULL,
  entity_type   text NOT NULL,
  entity_id     uuid,
  payload       jsonb,
  occurred_at   timestamptz NOT NULL DEFAULT now(),
  prev_hash     bytea,                     -- vorige rij's hash
  hash          bytea NOT NULL             -- SHA256(prev_hash || id || canonical_json(rij))
);
-- append-only enforcement via revoke UPDATE/DELETE en triggers

aangifte (Dimona / DmfA / Belcotax / Finprof)

CREATE TABLE aangifte (
  id            uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id     uuid NOT NULL REFERENCES tenant(id),
  werkgever_id  uuid NOT NULL,
  type          text NOT NULL CHECK (type IN ('dimona','dmfa','belcotax_281_10','belcotax_281_20','belcotax_281_25','belcotax_281_50','finprof')),
  periode       daterange,
  status        text NOT NULL CHECK (status IN ('voorbereid','verzonden','aanvaard','geweigerd','rectificatie')),
  payload       jsonb NOT NULL,            -- raw XML/JSON gemarshalled
  ticket_id     text,                       -- referentie aangeleverd door RSZ/FOD
  ingediend_op  timestamptz,
  response      jsonb
);

betaling (SEPA)

CREATE TABLE betaling (
  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id       uuid NOT NULL REFERENCES tenant(id),
  payrollrun_id   uuid REFERENCES payrollrun(id),
  begunstigde     text NOT NULL,
  iban_encrypted  bytea NOT NULL,
  bedrag          numeric(18,2) NOT NULL,
  munt            char(3) NOT NULL DEFAULT 'EUR',
  uitvoeringsdatum date NOT NULL,
  end_to_end_id   text UNIQUE NOT NULL,    -- voor reconciliatie
  pain001_batch_id uuid,
  status          text NOT NULL
);

4. Row-Level Security (multi-tenancy)

CREATE POLICY tenant_isolation ON werknemer
  USING (tenant_id = current_setting('app.current_tenant')::uuid);
